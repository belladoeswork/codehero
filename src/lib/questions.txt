    B) Math.max(...array)

    D) array.findMax()

    Question: How do you reverse a string in JavaScript?
    A) string.reverse()
    B) string.split('').reverse().join('')
    C) Array.reverse(string)
    D) string.split().reverse()
    Answer: B) string.split('').reverse().join('')


    Question: Which of the following functions correctly checks if a string is a palindrome?
    A) function isPalindrome(str) { return str === str.reverse(); }
    B) function isPalindrome(str) { return str === str.split('').reverse().join(''); }
    C) function isPalindrome(str) { return str.split(' ').reverse().join(' ') === str; }
    D) function isPalindrome(str) { return Array.reverse(str) === str; }
    Answer: B) function isPalindrome(str) { return str === str.split('').reverse().join(''); }

    Question: What does the following JavaScript function output for the number 15?

    function fizzBuzz(n) {
        for (let i = 1; i <= n; i++) {
            let output = '';
            if (i % 3 === 0) output += 'Fizz';
            if (i % 5 === 0) output += 'Buzz';
            console.log(output || i);
        }
    }

    A) Fizz
    B) Buzz
    C) FizzBuzz
    D) 15
    Answer: C) FizzBuzz

    Question: Which function counts the number of vowels in a string?
    A) function countVowels(str) { return str.match(/[aeiou]/gi).length; }
    B) function countVowels(str) { return str.split('aeiou').length - 1; }
    C) function countVowels(str) { return str.includes('a', 'e', 'i', 'o', 'u'); }
    D) function countVowels(str) { return str.count('aeiou'); }
    Answer: A) function countVowels(str) { return str.match(/[aeiou]/gi).length; }
